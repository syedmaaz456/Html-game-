  <div class="container">

      <h1>Welcome to the Puzzle Game</h1>

      <div class="level-grid" id="levelGrid"></div>

    </div>

 

    <div id="level" class="level">

      <h2 id="levelTitle">Level</h2>

      <p>Solve this simple puzzle: What is 2 + 2?</p>

      <input type="text" id="answer" placeholder="Enter your answer" />

      <button onclick="submitAnswer()">Submit</button>

      <p id="feedback"></p>

      <button class="back-button" onclick="goBack()">Back to Levels</button>

    </div>

  </div>

 

  <script>

    const levelGrid = document.getElementById("levelGrid");

    const home = document.getElementById("home");

    const level = document.getElementById("level");

    const levelTitle = document.getElementById("levelTitle");

    const feedback = document.getElementById("feedback");

    const answerInput = document.getElementById("answer");

    let currentLevel = 1;

 

    // Create 100 level buttons

    for (let i = 1; i <= 100; i++) {

      const btn = document.createElement("button");

      btn.className = "level-button";

      btn.textContent = `Level ${i}`;

      btn.onclick = () => startLevel(i);

      levelGrid.appendChild(btn);

    }

 

    function startLevel(levelNum) {

      currentLevel = levelNum;

      levelTitle.textContent = `Level ${levelNum}`;

      home.classList.remove("active");

      level.classList.add("active");

      feedback.textContent = "";

      answerInput.value = "";

    }

 

    function submitAnswer() {

      const userAnswer = answerInput.value.trim();

      if (userAnswer === "4") {

        feedback.textContent = `Correct! Level ${currentLevel} passed!`;

      } else {

        feedback.textContent = "Try again.";

      }

    }

 

    function goBack() {

      home.classList.add("active");

      level.classList.remove("active");

    }

  </script>
